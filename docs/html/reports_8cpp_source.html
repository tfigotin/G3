23{

40 // build left portion of the prompt (indent + text)
41 string prompt = "Enter Your Choice: ";
42 string leftPart = string(PROMPT_START_LENGTH, ' ') + prompt;
43
44 // Print the prompt (no newline) and flush so the user types on the same line.
45 cout << "*" << leftPart << flush;
46
47 // Read the user's line (skip leading whitespace/newline)
48 getline(cin >> ws, repChoice);
49
50 // Move the cursor up one line and to the start of that line, then rewrite the entire prompt line
51 // This overwrites the previously-echoed input line so the right '*' can be placed in the correct column.
52 cout << "\033[A\r";
53
54 // Make sure the displayed input doesn't overflow the inner width
55 int maxInputLen = INNER_WIDTH - (int)leftPart.length();
56 if (maxInputLen < 0) maxInputLen = 0;
57 string displayChoice = repChoice.substr(0, maxInputLen);
58
59 int used = (int)leftPart.length() + (int)displayChoice.length();
60 int spaces = INNER_WIDTH - used;
61 if (spaces < 0) spaces = 0;
62
63 cout << "*" << leftPart << displayChoice << string(spaces, ' ') << "*" << '\n';
64
65 // "You selected" line inside the same box
66 string selectedLine = "You selected: " + repChoice;
67 string leftSel = string(PROMPT_START_LENGTH, ' ') + selectedLine;
68 int usedSel = (int)leftSel.length();
69 int spacesSel = INNER_WIDTH - usedSel;
70 if (spacesSel < 0) {
71 // truncate if too long
72 int allowed = INNER_WIDTH - PROMPT_START_LENGTH;
73 if (allowed < 0) allowed = 0;
74 leftSel = string(PROMPT_START_LENGTH, ' ') + selectedLine.substr(0, allowed);
75 spacesSel = INNER_WIDTH - (int)leftSel.length();
76 }
77 cout << "*" << leftSel << string(spacesSel, ' ') << "*" << '\n';
78
79 printBorder();
80}
